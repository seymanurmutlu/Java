//bu program klavyeden girilen sayinin neon sayi olup olmadigini kontrol eder
public class deneme {
    
    public static void main (String[] args) {
        
        Scanner input = new Scanner(System.in);
        int sayi = input.nextInt();

        int i=0;
        double toplam=0;

        int kopya = sayi; //basamak sayisini bulurken sayiyi devamli bolecegimiz
        //icin sayinin kopyasini alidm

        int karsilastir = sayi;
        int basamakSayisi=0;
        while(sayi > 0){
            sayi = sayi/10;
            basamakSayisi++;
        }
        int[] dizi = new int[basamakSayisi]; /*girilen sayinin basamak say覺s覺 kadar dizi tan覺mlad覺m*/
	//cunku basamaklari diziye alicaz.
        
	System.out.println("basamak sayisi:"+basamakSayisi);

        while(kopya > 0){ //basamaklari teker teker dizi icerisine aliyoruz
            dizi[i] = kopya%10;
            kopya = kopya/10;
            i++;
        }
        
        for(i=0; i<basamakSayisi ; i++ ){
            System.out.println(dizi[i]);
            
            toplam = toplam + (double)dizi[i];//dizi int tipte oldugu icin casting yaptik
        }
        
            toplam = Math.pow(toplam,basamakSayisi);
            System.out.println(toplam);
        
            if(karsilastir == toplam)
            System.out.println("Neon sayi");
        
        else
            System.out.println("Neon sayi degil.");
}
  
    }
